110 template <typename T>
111 void display(std::pair<std::vector<T>, std::vector<std::vector<T>>> data,
112 bool display_all = false) {
113 int num_columns = data.first.size();
114 int num_rows = data.second.size();
115 int num_omitted_rows = 0;
116
117 for (const auto &header : data.first) {
118 std::cout << header << "\t";
119 }
120 std::cout << std::endl;
121
122 int num_elements = data.second.size();
123 if (!display_all && num_elements > 10) {
124 for (int i = 0; i < 5; i++) {
125 for (const auto &value : data.second[i]) {
126 std::cout << value << "\t";
127 }
128 std::cout << std::endl;
129 }
130 num_omitted_rows = num_elements - 10;
131 std::cout << "...\n";
132 std::cout << "[" << num_omitted_rows << " rows omitted]\n";
133 for (int i = num_elements - 5; i < num_elements; i++) {
134 for (const auto &value : data.second[i]) {
135 std::cout << value << "\t";
136 }
137 std::cout << std::endl;
138 }
139 } else {
140 for (const auto &row : data.second) {
141 for (const auto &value : row) {
142 std::cout << value << "\t";
143 }
144 std::cout << std::endl;
145 }
146 }
147
148 std::cout << "[" << num_rows << " rows"
149 << " x " << num_columns << " columns";
150 std::cout << "]\n\n";
151 }
152};
